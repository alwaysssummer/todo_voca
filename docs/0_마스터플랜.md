    swl.student_id,
    swl.wordlist_id,
    w.name as wordlist_name,
    w.total_words,

    w.wordlist_id,
    w.word_text,
    w.meaning,
    SUM(swp.skip_count) as total_skip_count,
    COUNT(DISTINCT swp.student_id) as student_count,
